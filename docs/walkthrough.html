// 2. Importation (on utilise 'THREE' pour correspondre à ton code)
THREE = require("three@0.160.0")
OrbitControls = {
  // On utilise la même logique que pour l'OBJLoader avec esm.sh
  const mod = await import("https://esm.sh/three@0.160.0/examples/jsm/controls/OrbitControls.js");
  return mod.OrbitControls; // On retourne la classe pour l'utiliser plus bas
}
OBJLoader = {
  const { OBJLoader } = await import("https://esm.sh/three@0.160.0/examples/jsm/loaders/OBJLoader.js");
  return new OBJLoader();
}
MTLLoader = {
  const { MTLLoader } = await import("https://esm.sh/three@0.160.0/examples/jsm/loaders/MTLLoader.js");
  return new MTLLoader();
}

GLTFLoader = {
  const { GLTFLoader } = await import("https://esm.sh/three@0.160.0/examples/jsm/loaders/GLTFLoader.js");
  return new GLTFLoader();
}



// 3. Création de la vue
canvas = {
  // On crée un conteneur HTML
  const container = html`<div style="height:${h}px; width:${w}px;"></div>`;
  
  const scene = new THREE.Scene();
  const camera = new THREE.PerspectiveCamera(60, w/h, 0.1, 100);
  camera.position.set(5, 5, 5);
  camera.lookAt(0, 0, 0); // On oriente la caméra vers le centre

  const renderer = new THREE.WebGLRenderer({ antialias: true });
  renderer.setSize(w, h);
  
  renderer.outputColorSpace = THREE.SRGBColorSpace;

  // On ajoute le canvas au conteneur
  container.appendChild(renderer.domElement);

  const controls = new OrbitControls(camera, renderer.domElement);
  controls.enableDamping = true; 
  controls.dampingFactor = 0.05;

  const size = 10;
  const divisions = 10;
  const gridHelper = new THREE.GridHelper(size, divisions, 0x888888, 0x444444);
  scene.add(gridHelper);

  // Optionnel : Un repère d'axes (Rouge=X, Vert=Y, Bleu=Z)
  const axesHelper = new THREE.AxesHelper(1);
  scene.add(axesHelper);

  // Ajout d'une lumière (indispensable pour le MeshStandardMaterial, sinon c'est noir !)
  const light = new THREE.DirectionalLight(0xffffff, 1);
  light.position.set(5, 5, 5);
  scene.add(light);
  scene.add(new THREE.AmbientLight(0x404040)); 

  const glbUrl = await FileAttachment("nounours2.glb").url();

  GLTFLoader.load(
    glbUrl, 
    (gltf) => {
      const model = gltf.scene;

      // Ajustement de la taille et de la position
      model.scale.set(0.1, 0.1, 0.1); 
      model.position.set(0, 0, 0);
      model.rotation.x = -Math.PI / 2;
      model.rotation.z = Math.PI / 2;

      scene.add(model);
    },
    (xhr) => { console.log('GLB: ' + (xhr.loaded / xhr.total * 100) + '%'); },
    (error) => { console.error('Erreur GLB', error); }
  );

  function animate() {
    controls.update();
    renderer.render(scene, camera);
    requestAnimationFrame(animate);
  }
  animate();

  // On retourne le conteneur pour qu'il s'affiche dans Quarto
  return container;
}
